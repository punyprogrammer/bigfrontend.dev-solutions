/**
 * Creates a debounced version of the input function `func`.
 * The debounced function delays invoking `func` until after `wait` milliseconds
 * have passed since the last time the debounced function was called.
 *
 * @param {Function} func - The function to debounce
 * @param {number} wait - The number of milliseconds to wait
 * @returns {Function} - A debounced version of `func`
 */
function debounce(func, wait) {
  // Holds the ID of the current timeout
  let timerId = null;

  // Return a new function that wraps the original `func`
  return function (...args) {
    // If there's an existing timer, cancel it (reset the countdown)
    if (timerId) clearTimeout(timerId);

    // Set a new timer that calls `func` after the specified `wait` time
    timerId = setTimeout(() => {
      // Call `func` with the correct `this` context and latest arguments
      func.apply(this, args);
    }, wait);
  };
